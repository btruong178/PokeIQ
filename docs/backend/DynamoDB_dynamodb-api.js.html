/**
 * @file
 * @description
 * Express router for DynamoDB endpoints.
 * @module DynamoDB-API
 * @see module:DynamoDB
 */
import { DynamoDBService } from "./dynamodb-service.js";
import { logError, logSuccess } from '../utils/Logger.js';
import express from 'express';

/**
 * Standard success response envelope returned by this router.
 * @typedef {Object} module:DynamoDB-API~ApiSuccess
 * @property {'success'} status - Indicates a successful request.
 * @property {string} [message] - Optional human-readable summary.
 * @property {*} [data] - Optional payload for non-item responses.
 * @property {*} [item] - Optional payload for item-based responses.
 */

/**
 * Standard error response envelope returned by this router.
 * @typedef {Object} module:DynamoDB-API~ApiError
 * @property {'error'} status - Indicates the request failed.
 * @property {string} message - Human-readable error summary.
 * @property {string} error - Error detail.
 */

// Initialize Clients
const dynamoDBService = new DynamoDBService();
const router = express.Router();

/**
 * Health-check endpoint handler.
 * HTTP: GET /health-check
 * @memberof module:DynamoDB-API
 * @param {Object} req - Express request object.
 * @param {Object} res - Express response object.
 * @returns {Promise<module:DynamoDB-API~ApiSuccess|module:DynamoDB-API~ApiError>} Sends the response.
 */
async function handleHealthCheck(req, res) {
    try {
        const result = await dynamoDBService.healthCheck();
        logSuccess('Health Check Endpoint', 'DynamoDB connection is healthy', result);
        return res.status(200).json({ status: 'success', message: 'Health check passed', data: result });
    } catch (error) {
        logError('Health Check Endpoint', error);
        return res.status(500).json({
            status: 'error',
            message: 'Health check failed',
            error: error.message
        });
    }
}


/**
 * Random Pokemon endpoint handler.
 * HTTP: GET /random-pokemon
 * Selects a random id from 1 to 1025 and returns that Pokemon item.
 * @memberof module:DynamoDB-API
 * @param {Object} req - Express request object.
 * @param {Object} res - Express response object.
 * @returns {Promise<module:DynamoDB-API~ApiSuccess|module:DynamoDB-API~ApiError>} Sends the response.
 */
async function handleRandomPokemon(req, res) {
    try {
        const randomID = Math.floor(Math.random() * 1025) + 1;
        const item = await dynamoDBService.getItem(process.env.DYNAMODB_TABLE_NAME_POKEMON, { id: randomID });
        logSuccess('Random Pokemon Endpoint', 'Fetched random Pokémon data', item);
        return res.status(200).json({ status: 'success', item });
    } catch (error) {
        logError('Random Pokemon Endpoint', error);
        return res.status(500).json({
            status: 'error',
            message: 'Failed to fetch random Pokémon data',
            error: error.message
        });
    }
}

/**
 * Damage-relations endpoint handler.
 * HTTP: GET /damage-relations/:type
 * Path params: type (Pokemon type name, case-insensitive).
 * @memberof module:DynamoDB-API
 * @param {Object} req - Express request object.
 * @param {Object} res - Express response object.
 * @returns {Promise<module:DynamoDB-API~ApiSuccess|module:DynamoDB-API~ApiError>} Sends the response.
 */
async function handleDamageRelations(req, res) {
    try {
        const type = req.params.type.toLowerCase();
        const item = await dynamoDBService.getItem(process.env.DYNAMODB_TABLE_NAME_TYPES, { name: type });
        return res.status(200).json({ status: 'success', item });
    } catch (error) {
        logError('Damage Relations Endpoint', error);
        return res.status(500).json({
            status: 'error',
            message: 'Failed to fetch damage relations data',
            error: error.message
        });
    }
}

router.get('/health-check', handleHealthCheck);
router.get('/random-pokemon', handleRandomPokemon);
router.get('/damage-relations/:type', handleDamageRelations);

export default router;
